Eliminar Un Archivo(LOC:FileName)

Agregar Un Archivo


        LOC:GitRep = 'D:/Multisoft/GitBash/ClarionSample'
        
        
        COPY(LOC:FileDir,LOC:GitRep)
        
        
        LOC:FileName = ST.FileNameOnly(LOC:FileDir) 
        
      
        Message(LOC:FileName)
        
      
        !RUN('git -C '&LOC:GitRep& ' add '&LOC:FileName)
        
        RUN('git -C '&LOC:GitRep& ' add '&LOC:FileName&'')
        
        LOC:CommitDesc = GitComment()
        
        MESSAGE(LOC:CommitDesc)
        
        !RUN('git -C '&LOC:GitRep&' commit -m "'&LOC:CommitDesc&'"')
        RUN('git -C '&LOC:GitRep&' commit -m "Ayo"')
        
        RUN('git -C '&LOC:GitRep&'/.git push')
